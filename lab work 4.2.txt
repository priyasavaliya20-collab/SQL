1️⃣ Products & Customers – All Combinations

CREATE TABLE products (
    product_id INT,
    product_name VARCHAR(50)
);

CREATE TABLE customers (
    customer_id INT,
    customer_name VARCHAR(50)
);

INSERT INTO products VALUES
(1, 'Laptop'),
(2, 'Mobile');

INSERT INTO customers VALUES
(101, 'Amit'),
(102, 'Neha');

SELECT p.product_name, c.customer_name
FROM products p
CROSS JOIN customers c;

QUESTION :- 2

1️⃣ Employees & Managers

CREATE TABLE emp_manager (
    emp_id INT,
    emp_name VARCHAR(50),
    manager_id INT
);

INSERT INTO emp_manager VALUES
(1, 'Amit', NULL),
(2, 'Neha', 1),
(3, 'Rahul', 1);

SELECT e.emp_name AS Employee, m.emp_name AS Manager
FROM emp_manager e
LEFT JOIN emp_manager m
ON e.manager_id = m.emp_id;

2️⃣ Customers Referring Customers

CREATE TABLE referral_customers (
    customer_id INT,
    customer_name VARCHAR(50),
    referrer_id INT
);

INSERT INTO referral_customers VALUES
(1, 'Amit', NULL),
(2, 'Neha', 1),
(3, 'Rahul', 1);

SELECT c.customer_name, r.customer_name AS Referred_By
FROM referral_customers c
LEFT JOIN referral_customers r
ON c.referrer_id = r.customer_id;

3️⃣ Organization Hierarchy

    parent_dept_id INT
);

INSERT INTO departments_hierarchy VALUES
(1, 'Head Office', NULL),

4️⃣ Connected Flights

CREATE TABLE flights (
    flight_id INT,
    source_city VARCHAR(50),
    destination_city VARCHAR(50)
);

INSERT INTO flights VALUES
(1, 'Delhi', 'Mumbai'),
(2, 'Mumbai', 'Chennai');

SELECT f1.source_city, f1.destination_city, f2.destination_city AS Connected_To
FROM flights f1
JOIN flights f2
ON f1.destination_city = f2.source_city;

5️⃣ Employees in Same Department

CREATE TABLE company_employees (
    emp_id INT,
    emp_name VARCHAR(50),
    dept_name VARCHAR(50)
);

INSERT INTO company_employees VALUES
(1, 'Amit', 'IT'),
(2, 'Neha', 'IT'),
(3, 'Rahul', 'HR');

SELECT e1.emp_name, e2.emp_name, e1.dept_name
FROM company_employees e1
JOIN company_employees e2
ON e1.dept_name = e2.dept_name
AND e1.emp_id <> e2.emp_id;

4️⃣ Custom

CREATE TABLE orders (
    order_id INT,
    customer_id INT,
    order_date DATE
);

INSERT INTO orders VALUES
(1, 101, '2024-01-01'),
(2, 101, '2024-02-01'),
(3, 102, '2024-03-01');

SELECT o1.customer_id, o1.order_date, o2.order_date AS Previous_Order
FROM orders o1
JOIN orders o2
ON o1.customer_id = o2.customer_id
AND o1.order_date > o2.order_date;

);
